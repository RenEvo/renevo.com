                var hascode = false;
                $("pre code").parent().each(function() {
                    if (!$(this).hasClass('prettyprint')) {
                        $(this).addClass('prettyprint');
                        hascode = true;
                    }
                });
                if (hascode) { prettyPrint(); }
            };
